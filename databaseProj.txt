CREATE DATABASE sitepomba;

USE sitepomba;

CREATE TABLE user(
iduser INT(6) AUTO_INCREMENT,
username VARCHAR(12) UNIQUE NOT NULL,
nome VARCHAR(100) NOT NULL,
email VARCHAR(100) NOT NULL,
type BOOLEAN NOT NULL DEFAULT 0,
senha CHAR(128) NOT NULL,
PRIMARY KEY (iduser)
);

CREATE TABLE post(
idpost INT(6) AUTO_INCREMENT,
titulo VARCHAR(200) NOT NULL,
subtitulo VARCHAR(200) NOT NULL,
postText TEXT NOT NULL,
postDia DATETIME DEFAULT CURRENT_TIMESTAMP,
img VARCHAR(50),
iduser INT(6) NOT NULL,
PRIMARY KEY (idpost),
FOREIGN KEY (iduser) REFERENCES user(iduser)
);

CREATE TABLE comment(
idcomment INT(6) AUTO_INCREMENT,
commentText TEXT NOT NULL,
commentDia DATETIME DEFAULT CURRENT_TIMESTAMP,
username VARCHAR(12) NOT NULL,
idpost INT(6) NOT NULL,
PRIMARY KEY (idcomment),
FOREIGN KEY (username) REFERENCES user(username),
FOREIGN KEY (idpost) REFERENCES post(idpost)
);

SELECT * FROM post RIGHT JOIN comment ON post.idpost = comment.idpost where post.idpost = 1;

INNER JOIN user ON comment.iduser = user.iduser
